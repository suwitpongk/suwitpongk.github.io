      @int_4 = int_4
    end
    
  def add

  end
    
  def multiply

  end
  
  def interpolated

  end
    
  def Divide
    sprintf "%.2f",  int_1.to_i/int_2.to_i
  end
  
  def Multiply
     sprintf "%.2f",  int_1.to_i*int_2.to_i
  end
